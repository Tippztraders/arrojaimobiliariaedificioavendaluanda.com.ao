function formatarMoeda(campo) {
  let valor = campo.value.replace(/\D/g, '');
  valor = (parseFloat(valor) / 100).toFixed(2) + '';
  valor = valor.replace('.', ',');
  valor = valor.replace(/\B(?=(\d{3})+(?!\d))/g, '.');
  campo.value = valor;
}
